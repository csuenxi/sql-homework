select xh,case
	when cj>=90 then'优'
	when cj between 80 and 90 then '良'
	when  cj<80 then '及格'
	end as 成绩
	from cj
	where kcdh='001'

select xs.xh,count(kcdh) 选课门数,case
	when count(kcdh)>4 then '多'
	when count(kcdh) between 2 and 4 then '一般'
	when  count(kcdh) between 1 and 2 then '少'
	when  count(kcdh) is null then '未选'
	end as 选课情况
	from cj right join xs on cj.xh=xs.xh
	where zy='应用电子技术'
	group by xs.xh
	order by count(kcdh) desc
