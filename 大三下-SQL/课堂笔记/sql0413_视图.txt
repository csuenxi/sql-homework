写出创建满足下述要求的视图的SQL语句，并执行所写代码
1、查询学生的学号，姓名，所在系，课程号，课程名，课程学分
2、查询学生的学号、姓名、选修的课程名和考试成绩
3、统计每个学生的选课门数，要求列出学号和选课门数
4、统计每个学生的选课总学分，列出学号和总学分（考试成绩大于60分才可以获取该门课程学分）
create view st1    
as
select SC.sno,sname,sdept,cname
from student,SC,course where SC.sno=student.sno
and course.cno=SC.cno

create view st2
as
select SC.sno,sname,cno,grade
from student,SC where SC.sno=student.sno

create view st3
as
select count(cno) 选课门数,sno
from SC
group by sno

create view st4
as
select sno,sum(SC.credit) 总学分 from SC
where grade>60
group by sno
